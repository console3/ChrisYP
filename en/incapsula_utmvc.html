Complete Process Example

import requests
from pynocaptcha import IncapsulaUtmvcCracker

def solve_incapsula_utmvc(target_url, proxy):
    """Complete Incapsula UTMVC solving process"""
    
    # 1. Request target page to get cookies
    session = requests.Session()
    session.proxies = {"http": proxy, "https": proxy}
    
    response = session.get(target_url)
    cookies = session.cookies.get_dict()
    
    # 2. Search for UTMVC script URL (containing SWJIYLWA keyword)
    # Need to extract actual script URL from page source
    utmvc_script_url = "https://premier.hkticketing.com/_Incapsula_Resource?SWJIYLWA=719d34d31c8e3a6e6fffd425f7e032f3"
    
    # 3. Get script content
    script_response = session.get(utmvc_script_url)
    script_content = script_response.text
    
    # 4. Call solving service
    cracker = IncapsulaUtmvcCracker(
        user_token="your_user_token_here",
        developer_id="hqLmMS",
        href=utmvc_script_url,
        script=script_content,
        cookies=cookies,
        proxy=proxy,
        user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/113.0.0.0 Safari/537.36"
    )
    
    result = cracker.crack()
    
    if result.get('status') == 1:
        utmvc_value = result['data']['___utmvc']
        print(f"Successfully obtained __utmvc: {utmvc_value}")
        return utmvc_value
    else:
        print(f"Solving failed: {result.get('msg')}")
        return None

# Usage example
utmvc = solve_incapsula_utmvc(
    target_url="https://premier.hkticketing.com/",
    proxy="usr:pwd@ip:port"
)
